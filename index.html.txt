<script>
    // ---------- 存储 ----------
    const STORAGE_KEY = 'repayment_flex_debt_v2';
    let records = [];      // [{ id, date, amount, note }]

    let repayChart = null;
    let forecastChart = null;

    // DOM 元素
    const dateInput = document.getElementById('repayDate');
    const amountInput = document.getElementById('repayAmount');
    const noteInput = document.getElementById('repayNote');
    const addBtn = document.getElementById('addRecordBtn');
    const remainingDebtInput = document.getElementById('currentRemainingDebt');
    const predictMethodSelect = document.getElementById('predictMethod');
    const totalRepaidSpanHero = document.getElementById('totalRepaidSpanHero');
    const cumulativeRepaySpan = document.getElementById('cumulativeRepaySpan');
    const highlightRemainingSpan = document.getElementById('highlightRemainingSpan');
    const predictionTextSpan = document.getElementById('predictionText');
    const recordsTbody = document.getElementById('recordsTbody');
    const monthlySummaryArea = document.getElementById('monthlySummaryArea');
    const exportCsvBtn = document.getElementById('exportCsvBtn');
    const exportExcelBtn = document.getElementById('exportExcelBtn');
    const resetDataBtn = document.getElementById('resetDataBtn');

    // 辅助: 默认日期今天
    function setDefaultDate() {
        if (!dateInput.value) {
            const today = new Date().toISOString().slice(0,10);
            dateInput.value = today;
        }
    }
    setDefaultDate();

    // 加载数据
    function loadRecords() {
        const stored = localStorage.getItem(STORAGE_KEY);
        if (stored) {
            try {
                records = JSON.parse(stored);
                records.forEach(r => { if(!r.id) r.id = Date.now() + Math.random(); if(r.note === undefined) r.note = ''; });
            } catch(e) { records = []; }
        } else {
            // 示例数据
            records = [
                { id: 'init1', date: '2026-03-05', amount: 1350, note: '部分还款' },
                { id: 'init2', date: '2026-03-18', amount: 2200, note: '奖金还款' },
                { id: 'init3', date: '2026-03-27', amount: 980, note: '额外还款' }
            ];
            saveRecords();
        }
        records.sort((a,b) => new Date(a.date) - new Date(b.date));
        // 加载剩余欠款存储
        const storedRemaining = localStorage.getItem('flex_remaining_debt_v2');
        if(storedRemaining !== null && !isNaN(parseFloat(storedRemaining))) {
            remainingDebtInput.value = parseFloat(storedRemaining);
        } else {
            // 基于示例动态计算一个合理的剩余欠款（假设原始欠款30000）
            const totalPaid = records.reduce((s,r)=>s+r.amount,0);
            const suggestRemaining = Math.max(5000, 30000 - totalPaid);
            remainingDebtInput.value = suggestRemaining.toFixed(2);
        }
        renderAll();
    }

    function saveRecords() {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(records));
        let rem = parseFloat(remainingDebtInput.value);
        if(!isNaN(rem)) localStorage.setItem('flex_remaining_debt_v2', rem);
    }

    // 获取当前剩余欠款 (核心)
    function getCurrentRemainingDebt() {
        let val = parseFloat(remainingDebtInput.value);
        return (isNaN(val) || val < 0) ? 0 : val;
    }

    // 累计还款总额
    function getTotalRepaid() {
        return records.reduce((sum, r) => sum + r.amount, 0);
    }

    // 平均还款额 (全部历史/近3次)
    function getAverageRepayment() {
        if(records.length === 0) return 0;
        const method = predictMethodSelect.value;
        if(method === 'avg_all') {
            const total = getTotalRepaid();
            return total / records.length;
        } else {
            const recent = records.slice(-3);
            const sumRecent = recent.reduce((s, r) => s + r.amount, 0);
            return sumRecent / recent.length;
        }
    }

    // 预测剩余还款次数
    function getPredictedRemainingTimes() {
        const remaining = getCurrentRemainingDebt();
        if(remaining <= 0) return 0;
        const avg = getAverageRepayment();
        if(avg <= 0 || records.length === 0) return Infinity;
        return Math.ceil(remaining / avg);
    }

    // 重点更新剩余欠款显示 + 预测UI
    function updateHighlightAndPrediction() {
        const remaining = getCurrentRemainingDebt();
        const totalPaid = getTotalRepaid();
        highlightRemainingSpan.innerHTML = `${remaining.toFixed(2)} <small>元</small>`;
        totalRepaidSpanHero.innerText = totalPaid.toFixed(2);
        cumulativeRepaySpan.innerText = totalPaid.toFixed(2);

        if(remaining <= 0) {
            predictionTextSpan.innerHTML = '🎉🎉 恭喜！剩余欠款为 0，你已经无债一身轻！太棒啦 🌟';
            return;
        }
        if(records.length === 0) {
            predictionTextSpan.innerHTML = '📌 暂无还款记录，添加几笔后，将根据平均还款额预测还清周期。';
            return;
        }
        const avg = getAverageRepayment();
        if(avg <= 0) {
            predictionTextSpan.innerHTML = '💚 请添加有效还款记录，以便预测。';
            return;
        }
        const times = getPredictedRemainingTimes();
        const methodName = predictMethodSelect.selectedOptions[0]?.text || '平均还款额';
        if(times === Infinity) {
            predictionTextSpan.innerHTML = '✨ 数据不足，继续记录还款即可看到预测。';
            return;
        }
        predictionTextSpan.innerHTML = `💪 当前剩余欠款 <strong style="font-size:1.1rem;">${remaining.toFixed(2)}</strong> 元 · 基于“${methodName}” 单次约 ${avg.toFixed(2)} 元<br> 🕊️ 预计还需 <strong>${times}</strong> 次还款即可还清 (保持当前趋势)`;
    }

    // 渲染双图表
    function renderCharts() {
        const totalRepaid = getTotalRepaid();
        if(records.length === 0) {
            if(repayChart) repayChart.destroy();
            if(forecastChart) forecastChart.destroy();
            const ctx1 = document.getElementById('repayChart').getContext('2d');
            const ctx2 = document.getElementById('forecastChart').getContext('2d');
            repayChart = new Chart(ctx1, { type: 'bar', data: { labels: [], datasets: [] }, options: { responsive: true } });
            forecastChart = new Chart(ctx2, { type: 'line', data: { labels: [], datasets: [] } });
            updateHighlightAndPrediction();
            return;
        }

        const sorted = [...records].sort((a,b)=> new Date(a.date) - new Date(b.date));
        const labels = sorted.map(r => r.date.slice(5));
        const amounts = sorted.map(r => r.amount);
        let cumulative = [];
        let sum = 0;
        for(let amt of amounts) { sum += amt; cumulative.push(sum); }

        // 图表1: 历史还款
        const ctx1 = document.getElementById('repayChart').getContext('2d');
        if(repayChart) repayChart.destroy();
        repayChart = new Chart(ctx1, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [
                    { label: '单次还款金额 (元)', data: amounts, backgroundColor: '#7bc49a', borderRadius: 8, yAxisID: 'y' },
                    { label: '累计还款总额 (元)', data: cumulative, type: 'line', borderColor: '#2a6e4e', borderWidth: 3, fill: false, tension: 0.2, pointBackgroundColor: '#175e3f', yAxisID: 'y1' }
                ]
            },
            options: {
                responsive: true, maintainAspectRatio: true,
                plugins: { tooltip: { mode: 'index', intersect: false }, legend: { position: 'top' } },
                scales: { y: { title: { display: true, text: '单次金额 (¥)' }, beginAtZero: true }, y1: { position: 'right', title: { text: '累计 (¥)' }, grid: { drawOnChartArea: false }, beginAtZero: true } }
            }
        });

        // 图表2: 预测趋势 (累计还款 + 还清目标线)
        const remainingNow = getCurrentRemainingDebt();
        const totalPaidNow = totalRepaid;
        const avgVal = getAverageRepayment();
        const targetTotal = totalPaidNow + remainingNow;  // 还清所需累计总额
        
        let historicalCumulative = cumulative;
        let futureLabels = [...labels];
        let futureData = [...historicalCumulative];
        
        if(remainingNow > 0 && avgVal > 0 && records.length > 0) {
            let tempSum = totalPaidNow;
            let steps = 0;
            const maxSteps = 35;
            while(tempSum < targetTotal && steps < maxSteps) {
                tempSum += avgVal;
                steps++;
                if(tempSum > targetTotal) tempSum = targetTotal;
                futureData.push(tempSum);
                futureLabels.push(`预测${steps}`);
                if(tempSum >= targetTotal) break;
            }
        }
        
        const ctx2 = document.getElementById('forecastChart').getContext('2d');
        if(forecastChart) forecastChart.destroy();
        forecastChart = new Chart(ctx2, {
            type: 'line',
            data: {
                labels: futureLabels,
                datasets: [
                    {
                        label: '累计还款金额 (历史 + 预测趋势)',
                        data: futureData,
                        borderColor: '#2f8b60',
                        backgroundColor: 'rgba(47, 139, 96, 0.08)',
                        borderWidth: 3,
                        fill: true,
                        tension: 0.2,
                        pointRadius: 3
                    },
                    {
                        label: `🎯 还清目标线 (累计需达 ${targetTotal.toFixed(2)} 元)`,
                        data: new Array(futureLabels.length).fill(targetTotal),
                        borderColor: '#cf8b72',
                        borderWidth: 2.5,
                        borderDash: [8, 6],
                        fill: false,
                        pointRadius: 0
                    }
                ]
            },
            options: {
                responsive: true,
                plugins: { tooltip: { callbacks: { label: (ctx) => `${ctx.dataset.label}: ${ctx.raw.toFixed(2)} 元` } } },
                scales: { y: { title: { display: true, text: '金额 (元)' }, beginAtZero: true } }
            }
        });
        updateHighlightAndPrediction();
    }

    // 月度汇总 + 表格
    function renderMonthlyAndTable() {
        const monthMap = new Map();
        records.forEach(rec => {
            const ym = rec.date.slice(0,7);
            monthMap.set(ym, (monthMap.get(ym) || 0) + rec.amount);
        });
        const sortedMonths = Array.from(monthMap.keys()).sort();
        let monthlyHtml = `<div style="display:flex; flex-wrap:wrap; gap:0.8rem;">`;
        if(sortedMonths.length === 0) monthlyHtml += `<span style="color:#789f8a;">暂无还款记录~</span>`;
        sortedMonths.forEach(ym => {
            monthlyHtml += `<div style="background:#e7f3ed; border-radius:1.5rem; padding:0.3rem 1rem;"><span style="font-weight:500;">${ym}</span> 共 ¥${monthMap.get(ym).toFixed(2)}</div>`;
        });
        monthlyHtml += `</div>`;
        monthlySummaryArea.innerHTML = monthlyHtml;
        
        if(records.length === 0) {
            recordsTbody.innerHTML = '<tr><td colspan="4" style="text-align:center;">🌱 点击上方记录第一笔还款吧</td></tr>';
            return;
        }
        let htmlRows = '';
        records.forEach(rec => {
            htmlRows += `
                <tr>
                    <td>${rec.date}</td>
                    <td>¥ ${rec.amount.toFixed(2)}</td>
                    <td>${escapeHtml(rec.note || '')}</td>
                    <td><button class="delete-btn" data-id="${rec.id}">🗑️</button></td>
                </tr>
            `;
        });
        recordsTbody.innerHTML = htmlRows;
        document.querySelectorAll('.delete-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = btn.getAttribute('data-id');
                deleteRecordById(id);
            });
        });
    }
    
    function escapeHtml(str) { if(!str) return ''; return str.replace(/[&<>]/g, function(m){if(m==='&') return '&amp;'; if(m==='<') return '&lt;'; if(m==='>') return '&gt;'; return m;}); }
    
    function deleteRecordById(id) {
        records = records.filter(r => r.id != id);
        saveRecords();
        renderAll();
    }
    
    function addNewRecord() {
        let date = dateInput.value;
        let amount = parseFloat(amountInput.value);
        let note = noteInput.value.trim();
        if(!date) { alert('请选择还款日期'); return; }
        if(isNaN(amount) || amount <= 0) { alert('请输入有效的还款金额 (>0)'); return; }
        if(note === '') note = '无备注';
        const newId = Date.now() + '' + Math.floor(Math.random()*100000);
        records.push({ id: newId, date: date, amount: amount, note: note });
        records.sort((a,b)=> new Date(a.date) - new Date(b.date));
        saveRecords();
        amountInput.value = '';
        noteInput.value = '';
        setDefaultDate();
        renderAll();
    }
    
    function renderAll() {
        renderMonthlyAndTable();
        renderCharts();
    }
    
    function bindEvents() {
        addBtn.addEventListener('click', addNewRecord);
        remainingDebtInput.addEventListener('input', () => {
            saveRecords();
            renderCharts();
            renderMonthlyAndTable();
        });
        predictMethodSelect.addEventListener('change', () => { renderCharts(); });
        exportCsvBtn.addEventListener('click', exportToCSV);
        exportExcelBtn.addEventListener('click', exportToExcel);
        resetDataBtn.addEventListener('click', () => {
            if(confirm('⚠️ 重置将清空所有还款记录，并重置剩余欠款为默认值，是否继续？')) {
                records = [];
                saveRecords();
                remainingDebtInput.value = 25000;
                localStorage.setItem('flex_remaining_debt_v2', 25000);
                renderAll();
            }
        });
    }
    
    function exportToCSV() {
        let content = "日期,金额(元),备注\n";
        records.forEach(r => {
            content += `${r.date},${r.amount},"${(r.note || '').replace(/"/g, '""')}"\n`;
        });
        const blob = new Blob(["\uFEFF" + content], {type: "text/csv;charset=utf-8;"});
        const link = document.createElement('a');
        const url = URL.createObjectURL(blob);
        link.href = url;
        link.download = "还款记录.csv";
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
    }
    
    function exportToExcel() {
        const sheetData = records.map(r => ({ '日期': r.date, '还款金额(元)': r.amount, '备注': r.note || '' }));
        const ws = XLSX.utils.json_to_sheet(sheetData);
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, "还款记录");
        XLSX.writeFile(wb, `还款明细_${new Date().toISOString().slice(0,10)}.xlsx`);
    }
    
    loadRecords();
    bindEvents();
    window.addEventListener('load', () => { renderAll(); });
